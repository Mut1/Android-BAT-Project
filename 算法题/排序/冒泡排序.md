```java
protected void Bubblesort(Integer[] array) {
		for (int end = array.length - 1; end > 0; end--) {
			for (int begin = 1; begin <= end; begin++) {
				 if (array[begin] < array[begin - 1]) {
					swap(begin, begin - 1);
				}
			}
		}
	}
```

- 如果序列已经完全有序，可以提前终止冒泡排序

```java
	protected void Bullesort2(Integer[] array) {
		for (int end = array.length - 1; end > 0; end--) {
			boolean sorted = true;
			for (int begin = 1; begin <= end; begin++) {
			  if (array[begin] < array[begin - 1]) {
					swap(begin, begin - 1);
					sorted = false;
				}
			}
			if (sorted) break;
		}
	}
```
